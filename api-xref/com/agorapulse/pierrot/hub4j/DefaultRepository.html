18  package com.agorapulse.pierrot.hub4j;
19  
20  import com.agorapulse.pierrot.api.Content;
21  import com.agorapulse.pierrot.api.GitHubConfiguration;
22  import com.agorapulse.pierrot.api.PullRequest;
23  import com.agorapulse.pierrot.api.Repository;
24  import com.agorapulse.pierrot.api.util.LoggerWithOptionalStacktrace;
25  import io.micronaut.http.client.HttpClient;
26  import org.kohsuke.github.GHContent;
27  import org.kohsuke.github.GHContentBuilder;
28  import org.kohsuke.github.GHIssueState;
29  import org.kohsuke.github.GHPermissionType;
30  import org.kohsuke.github.GHPullRequest;
31  import org.kohsuke.github.GHRepository;
32  import org.kohsuke.github.GHUser;
33  import org.kohsuke.github.PagedIterator;
34  import org.slf4j.Logger;
35  
36  import java.io.IOException;
37  import java.util.EnumSet;
38  import java.util.Optional;
39  
40  public class DefaultRepository implements Repository {
41  
42      // the field is not static to prevent GraalVM FileAppender issues
43      private static final Logger LOGGER = LoggerWithOptionalStacktrace.create(DefaultRepository.class);
44      private static final EnumSet<GHPermissionType> WRITE_PERMISSIONS = EnumSet.of(GHPermissionType.WRITE, GHPermissionType.ADMIN);
45  
46      private final GHRepository repository;
47      private final GHUser myself;
48      private final GitHubConfiguration configuration;
49      private final HttpClient client;
50  
51      public DefaultRepository(GHRepository repository, GHUser myself, GitHubConfiguration configuration, HttpClient client) {
52          this.repository = repository;
53          this.myself = myself;
54          this.configuration = configuration;
55          this.client = client;
56      }
57  
58      @Override
59      public String getFullName() {
60          return repository.getFullName();
61      }
62  
63      @Override
64      public String getName() {
65          return repository.getName();
66      }
67  
68      @Override
69      public String getOwnerName() {
70          return repository.getOwnerName();
71      }
72  
73      @Override
74      public boolean isArchived() {
75          return repository.isArchived();
76      }
77  
78      @Override
79      public boolean canWrite() {
80          try {
81              return WRITE_PERMISSIONS.contains(repository.getPermission(myself));
82          } catch (IOException e) {
83              LOGGER.info("Exception evaluating permissions for {}", getFullName());
84              return false;
85          }
86      }
87  
88      @Override
89      public boolean createBranch(String name, boolean force) {
90          if (hasBranch(name)) {
91              LOGGER.info("Branch {} already exists in repository {}", name, getFullName());
92              if (!force) {
93                  return false;
94              }
95              deleteBranch(name);
96              LOGGER.info("Existing branch {} has been deleted in repository {}", name, getFullName());
97          }
98  
99          try {
100             repository.createRef("refs/heads/" + name, getLastCommitSha());
101             LOGGER.info("Branch {} created in repository {}", name, getFullName());
102             return true;
103         } catch (IOException e) {
104             LOGGER.error("Exception creating branch " + name, e);
105             return false;
106         }
107     }
108 
109     @Override
110     public Optional<PullRequest> createPullRequest(String branch, String title, String message) {
111         try {
112             PagedIterator<GHPullRequest> iterator = repository.queryPullRequests().head(branch).base(getDefaultBranch()).list().iterator();
113             if (iterator.hasNext()) {
114                 GHPullRequest existing = iterator.next();
115                 if (!GHIssueState.OPEN.equals(existing.getState())) {
116                     existing.reopen();
117                 }
118                 return Optional.of(new DefaultPullRequest(
119                     existing,
120                     repository,
121                     myself,
122                     configuration,
123                     client));
124             }
125             return Optional.of(new DefaultPullRequest(
126                 repository.createPullRequest(title, branch, getDefaultBranch(), message),
127                 repository,
128                 myself,
129                 configuration,
130                 client));
131         } catch (IOException e) {
132             LOGGER.error("Exception creating pull request " + title, e);
133             return Optional.empty();
134         }
135     }
136 
137     @Override
138     public boolean writeFile(String branch, String message, String path, String text) {
139         try {
140             GHContentBuilder builder = repository.createContent().branch(branch).path(path).content(text).message(message);
141             Optional<Content> existing = getFile(branch, path);
142             if (existing.isPresent()) {
143                 Content content = existing.get();
144                 if (text.equals(content.getTextContent())) {
145                     LOGGER.info("Remote file {} already contains all the changes", path);
146                     return false;
147                 }
148                 builder.sha(content.getSha());
149             }
150             builder.commit();
151             LOGGER.info("File {} pushed to branch {} of repository {}", path, branch, getFullName());
152             return true;
153         } catch (IOException e) {
154             LOGGER.error("Exception writing file " + path, e);
155             return false;
156         }
157     }
158 
159     private void deleteBranch(String name) {
160         try {
161             repository.getRef("heads/" + name).delete();
162         } catch (IOException e) {
163             LOGGER.error("Exception deleting branch " + name + " in " + getName(), e);
164         }
165     }
166 
167     private Optional<Content> getFile(String branch, String path) {
168         try {
169             GHContent fileContent = repository.getFileContent(path, branch);
170             return Optional.of(new DefaultContent(fileContent, repository, myself, configuration, client));
171         } catch (IOException e) {
172             LOGGER.error("Exception fetching file " + path, e);
173             return Optional.empty();
174         }
175     }
176 
177     private String getDefaultBranch() {
178         return repository.getDefaultBranch() == null ? configuration.getDefaultBranch() : repository.getDefaultBranch();
179     }
180 
181     private boolean hasBranch(String name) {
182         try {
183             return repository.getBranches().containsKey(name);
184         } catch (IOException e) {
185             LOGGER.error("Exception checking presence of branch " + name, e);
186             return false;
187         }
188     }
189 
190     private String getLastCommitSha() {
191         return repository.listCommits().iterator().next().getSHA1();
192     }
193 
194 }
