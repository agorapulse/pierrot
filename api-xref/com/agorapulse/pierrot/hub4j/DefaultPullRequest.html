18  package com.agorapulse.pierrot.hub4j;
19  
20  import com.agorapulse.pierrot.api.CheckRun;
21  import com.agorapulse.pierrot.api.GitHubConfiguration;
22  import com.agorapulse.pierrot.api.PullRequest;
23  import com.agorapulse.pierrot.api.Repository;
24  import com.agorapulse.pierrot.api.util.LoggerWithOptionalStacktrace;
25  import io.micronaut.context.event.ApplicationEventPublisher;
26  import io.micronaut.core.util.StringUtils;
27  import io.micronaut.http.HttpRequest;
28  import io.micronaut.http.MutableHttpRequest;
29  import io.micronaut.http.client.HttpClient;
30  import io.micronaut.http.client.exceptions.HttpClientResponseException;
31  import org.kohsuke.github.GHIssueState;
32  import org.kohsuke.github.GHPullRequest;
33  import org.kohsuke.github.GHRepository;
34  import org.kohsuke.github.GHUser;
35  import org.slf4j.Logger;
36  
37  import java.io.IOException;
38  import java.net.URL;
39  import java.util.stream.Stream;
40  import java.util.stream.StreamSupport;
41  
42  public class DefaultPullRequest implements PullRequest {
43  
44      // the field is not static to prevent GraalVM FileAppender issues
45      private static final Logger LOGGER = LoggerWithOptionalStacktrace.create(DefaultPullRequest.class);
46  
47      private final GHPullRequest pr;
48      private final GHRepository repository;
49      private final GHUser myself;
50      private final GitHubConfiguration configuration;
51      private final HttpClient client;
52      private final ApplicationEventPublisher publisher;
53  
54      public DefaultPullRequest(GHPullRequest pr, GHRepository repository, GHUser myself, GitHubConfiguration configuration, HttpClient client, ApplicationEventPublisher publisher) {
55          this.pr = pr;
56          this.repository = repository;
57          this.myself = myself;
58          this.configuration = configuration;
59          this.client = client;
60          this.publisher = publisher;
61      }
62  
63      @Override
64      public Repository getRepository() {
65          return new DefaultRepository(repository, myself, configuration, client, publisher);
66      }
67  
68      @Override
69      public String getTitle() {
70          return pr.getTitle();
71      }
72  
73      @Override
74      public String getBody() {
75          return pr.getBody();
76      }
77  
78      @Override
79      public boolean isMerged() {
80          try {
81              return pr.isMerged();
82          } catch (IOException e) {
83              LOGGER.error("Exception fetching merged state", e);
84              return false;
85          }
86      }
87  
88      @Override
89      public boolean isMergeable() {
90          try {
91              return Boolean.TRUE.equals(pr.getMergeable());
92          } catch (IOException e) {
93              LOGGER.error("Exception fetching mergeable state", e);
94              return false;
95          }
96      }
97  
98      @Override
99      public String getMergeableState() {
100         try {
101             return pr.getMergeableState();
102         } catch (IOException e) {
103             LOGGER.error("Exception fetching mergeable state", e);
104             return "unknown";
105         }
106     }
107 
108     @Override
109     public Stream<? extends CheckRun> getChecks() {
110         try {
111             return StreamSupport.stream(repository.getCommit(pr.getHead().getSha()).getCheckRuns().spliterator(), false).map(DefaultCheckRun::new);
112         } catch (IOException e) {
113             LOGGER.error("Exception fetching check runs", e);
114             return Stream.empty();
115         }
116     }
117 
118     @Override
119     public URL getHtmlUrl() {
120         return pr.getHtmlUrl();
121     }
122 
123     @Override
124     public boolean close(boolean delete) {
125         String repoName = pr.getRepository().getFullName();
126 
127         if (GHIssueState.CLOSED.equals(pr.getState())) {
128             return true;
129         }
130 
131         try {
132             pr.close();
133             LOGGER.info("Closed #{}/{}: {}", repoName, pr.getId(), pr.getTitle());
134         } catch (IOException e) {
135             LOGGER.error(String.format("Exception closing pull request for #%d: %s", pr.getId(), pr.getTitle()), e);
136             return false;
137         }
138 
139         if (!delete) {
140             return false;
141         }
142 
143         String branchName = pr.getHead().getRef();
144 
145         if (StringUtils.isEmpty(configuration.getToken())) {
146             LOGGER.error("Missing token, cannot delete pull request branch {} for #{}: {}", branchName, pr.getId(), pr.getTitle());
147             return false;
148         }
149 
150         MutableHttpRequest<Object> request = HttpRequest
151             .DELETE("https://api.github.com/repos/" + repoName + "/git/refs/heads/" + branchName)
152             .header("User-Agent", "Pierrot")
153             .header("Authorization", "token " + configuration.getToken());
154 
155         try {
156             client.toBlocking().exchange(request);
157             LOGGER.info("Deleted branch {} in {}", branchName, repoName);
158             return true;
159         } catch (HttpClientResponseException e) {
160             LOGGER.error(String.format("Cannot delete pull request branch %s for #%s: %s", branchName, pr.getId(), pr.getTitle()), e);
161             return false;
162         }
163     }
164 
165     GHPullRequest getNativePullRequest() {
166         return pr;
167     }
168 }
