18  package com.agorapulse.pierrot.cli;
19  
20  import com.agorapulse.pierrot.api.GitHubService;
21  import com.agorapulse.pierrot.api.ws.Workspace;
22  import com.agorapulse.pierrot.cli.mixin.SearchMixin;
23  import com.agorapulse.pierrot.cli.mixin.WorkspaceMixin;
24  import jakarta.inject.Inject;
25  import picocli.CommandLine;
26  import picocli.CommandLine.Command;
27  import picocli.CommandLine.Mixin;
28  
29  import java.io.File;
30  import java.util.ArrayList;
31  import java.util.List;
32  import java.util.Optional;
33  
34  @Command(
35      name = "pull",
36      description = "pulls the matching files locally",
37      mixinStandardHelpOptions = true
38  )
39  public class PullCommand implements Runnable {
40  
41      @Mixin SearchMixin search;
42      @Mixin WorkspaceMixin workspace;
43  
44      @CommandLine.Option(
45          names = {"-o", "--workspace-repositories-only"},
46          description = "Search only within the existing workspace repositories"
47      )
48      boolean workspaceRepositoriesOnly;
49  
50      @Inject GitHubService service;
51  
52      @Override
53      public void run() {
54          if (workspaceRepositoriesOnly) {
55              Workspace w = new Workspace(workspace.getWorkspace());
56              List<String> repositoryFullNames = new ArrayList<>();
57              w.visitRepositories(r -> repositoryFullNames.add(r.getName()));
58              search.constraintToRepositories(repositoryFullNames);
59          }
60  
61          search.searchContent(service, content -> {
62              File location = new File(workspace.getWorkspace(), String.format("%s/%s", content.getRepository().getFullName(), content.getPath()));
63              content.writeTo(location);
64              System.out.printf("Fetched %s/%s%n", content.getRepository().getFullName(), content.getPath());
65              return Optional.of(location.toURI());
66          });
67  
68          System.out.printf("Found %d results!%n", search.getProcessed());
69      }
70  }
